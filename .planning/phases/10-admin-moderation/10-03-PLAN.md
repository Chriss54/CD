import { canManageMembers, canManageRole, canAssignRole, isValidRole, type Role } from '@/lib/permissions';

// Ban user temporarily (admin+)
export async function banUser(
  userId: string,
  data: { reason: string; durationDays: 1 | 7 | 30 }
) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.id) return { error: 'Not authenticated' };

  const actorRole = session.user.role as Role;
  if (!canManageMembers(actorRole)) {
    return { error: 'Not authorized - admin role required' };
  }

  const targetUser = await db.user.findUnique({
    where: { id: userId },
    select: { id: true, role: true },
  });

  if (!targetUser) return { error: 'User not found' };

  // Cannot ban users at same or higher level
  if (!canManageRole(actorRole, targetUser.role as Role)) {
    return { error: 'Cannot ban user with equal or higher role' };
  }

  const expiresAt = new Date();
  expiresAt.setDate(expiresAt.getDate() + data.durationDays);

  await db.$transaction(async (tx) => {
    // Create ban record
    await tx.ban.create({
      data: {
        userId,
        reason: data.reason,
        expiresAt,
        bannedById: session.user.id,
      },
    });

    // Delete user's content (hard delete per CONTEXT.md)
    await tx.commentLike.deleteMany({ where: { userId } });
    await tx.postLike.deleteMany({ where: { userId } });
    await tx.comment.deleteMany({ where: { authorId: userId } });
    await tx.post.deleteMany({ where: { authorId: userId } });

    // Log audit event
    await tx.auditLog.create({
      data: {
        actorId: session.user.id,
        action: 'MEMBER_BANNED',
        targetType: 'user',
        targetId: userId,
        targetOwnerId: userId,
        metadata: {
          reason: data.reason,
          durationDays: data.durationDays,
          expiresAt: expiresAt.toISOString(),
        },
      },
    });
  });

  revalidatePath('/admin/members');
  revalidatePath('/feed');
  return { success: true };
}

// Remove user permanently (admin+)
export async function removeUser(userId: string, reason: string) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.id) return { error: 'Not authenticated' };

  const actorRole = session.user.role as Role;
  if (!canManageMembers(actorRole)) {
    return { error: 'Not authorized - admin role required' };
  }

  const targetUser = await db.user.findUnique({
    where: { id: userId },
    select: { id: true, role: true, email: true },
  });

  if (!targetUser) return { error: 'User not found' };

  // Cannot remove users at same or higher level
  if (!canManageRole(actorRole, targetUser.role as Role)) {
    return { error: 'Cannot remove user with equal or higher role' };
  }

  await db.$transaction(async (tx) => {
    // Log audit event first (before deletion)
    await tx.auditLog.create({
      data: {
        actorId: session.user.id,
        action: 'MEMBER_REMOVED',
        targetType: 'user',
        targetId: userId,
        targetOwnerId: userId,
        metadata: {
          reason,
          email: targetUser.email, // Preserve for records
        },
      },
    });

    // Delete user (cascade will handle content)
    await tx.user.delete({ where: { id: userId } });
  });

  revalidatePath('/admin/members');
  revalidatePath('/feed');
  return { success: true };
}

// Change user role (admin+ for mod, owner for admin)
export async function changeUserRole(userId: string, newRole: string) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.id) return { error: 'Not authenticated' };

  if (!isValidRole(newRole)) {
    return { error: 'Invalid role' };
  }

  const actorRole = session.user.role as Role;
  if (!canAssignRole(actorRole, newRole as Role)) {
    return { error: 'Not authorized to assign this role' };
  }

  const targetUser = await db.user.findUnique({
    where: { id: userId },
    select: { id: true, role: true },
  });

  if (!targetUser) return { error: 'User not found' };

  // Cannot change role of users at same or higher level (except own)
  if (userId !== session.user.id && !canManageRole(actorRole, targetUser.role as Role)) {
    return { error: 'Cannot modify user with equal or higher role' };
  }

  const oldRole = targetUser.role;

  await db.user.update({
    where: { id: userId },
    data: { role: newRole },
  });

  // Log audit event
  await logAuditEvent({
    actorId: session.user.id,
    action: 'MEMBER_ROLE_CHANGED',
    targetType: 'user',
    targetId: userId,
    metadata: {
      oldRole,
      newRole,
    },
  });

  revalidatePath('/admin/members');
  return { success: true };
}

// Get members for admin (with ban status)
export async function getMembersForAdmin(page: number = 1, pageSize: number = 20) {
  const skip = (page - 1) * pageSize;
  const now = new Date();

  const [users, total] = await Promise.all([
    db.user.findMany({
      skip,
      take: pageSize,
      orderBy: { createdAt: 'desc' },
      select: {
        id: true,
        name: true,
        email: true,
        image: true,
        role: true,
        points: true,
        level: true,
        createdAt: true,
        bans: {
          where: { expiresAt: { gt: now } },
          orderBy: { expiresAt: 'desc' },
          take: 1,
          select: {
            id: true,
            reason: true,
            expiresAt: true,
          },
        },
      },
    }),
    db.user.count(),
  ]);

  return {
    users: users.map((user) => ({
      ...user,
      activeBan: user.bans[0] ?? null,
      bans: undefined,
    })),
    total,
    page,
    pageSize,
    totalPages: Math.ceil(total / pageSize),
  };
}

// Check if user is banned (utility for auth)
export async function isUserBanned(userId: string): Promise<{
  banned: boolean;
  reason?: string;
  expiresAt?: Date;
}> {
  const ban = await db.ban.findFirst({
    where: {
      userId,
      expiresAt: { gt: new Date() },
    },
    orderBy: { expiresAt: 'desc' },
    select: { reason: true, expiresAt: true },
  });

  if (ban) {
    return { banned: true, reason: ban.reason, expiresAt: ban.expiresAt };
  }
  return { banned: false };
}

'use client';

import { useState } from 'react';
import { Button } from '@/components/ui/button';
import { banUser } from '@/lib/admin-actions';
import { toast } from 'sonner';

interface BanDialogProps {
  userId: string;
  userName: string;
  onClose: () => void;
}

const DURATION_OPTIONS = [
  { value: 1, label: '1 day' },
  { value: 7, label: '7 days' },
  { value: 30, label: '30 days' },
] as const;

export function BanDialog({ userId, userName, onClose }: BanDialogProps) {
  const [reason, setReason] = useState('');
  const [duration, setDuration] = useState<1 | 7 | 30>(7);
  const [isSubmitting, setIsSubmitting] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!reason.trim()) {
      toast.error('Ban reason is required');
      return;
    }

    setIsSubmitting(true);
    const result = await banUser(userId, { reason: reason.trim(), durationDays: duration });
    setIsSubmitting(false);

    if (result.error) {
      toast.error(result.error);
    } else {
      toast.success(`${userName} has been banned for ${duration} days`);
      onClose();
    }
  };

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center">
      <div className="absolute inset-0 bg-black/50" onClick={onClose} />
      <div className="relative bg-background rounded-lg shadow-lg p-6 w-full max-w-md mx-4">
        <h2 className="text-lg font-semibold mb-4">Ban {userName}</h2>
        <form onSubmit={handleSubmit} className="space-y-4">
          <div>
            <label className="block text-sm font-medium mb-1">
              Duration
            </label>
            <div className="flex gap-2">
              {DURATION_OPTIONS.map((opt) => (
                <button
                  key={opt.value}
                  type="button"
                  onClick={() => setDuration(opt.value)}
                  className={`px-3 py-1.5 text-sm rounded-md border ${
                    duration === opt.value
                      ? 'bg-primary text-primary-foreground border-primary'
                      : 'bg-background border-input hover:bg-muted'
                  }`}
                >
                  {opt.label}
                </button>
              ))}
            </div>
          </div>
          <div>
            <label className="block text-sm font-medium mb-1">
              Reason <span className="text-destructive">*</span>
            </label>
            <textarea
              value={reason}
              onChange={(e) => setReason(e.target.value)}
              placeholder="Explain why this user is being banned..."
              className="w-full px-3 py-2 border rounded-md text-sm min-h-[100px] resize-none"
              maxLength={500}
              required
            />
            <p className="text-xs text-muted-foreground mt-1">
              The user will see this reason.
            </p>
          </div>
          <div className="flex justify-end gap-2 pt-2">
            <Button type="button" variant="outline" onClick={onClose}>
              Cancel
            </Button>
            <Button type="submit" variant="destructive" disabled={isSubmitting}>
              {isSubmitting ? 'Banning...' : 'Ban User'}
            </Button>
          </div>
        </form>
      </div>
    </div>
  );
}

'use client';

import { useState } from 'react';
import Image from 'next/image';
import { formatDistanceToNow } from 'date-fns';
import { Button } from '@/components/ui/button';
import { RoleBadge } from '@/components/admin/role-badge';
import { BanDialog } from '@/components/admin/ban-dialog';
import { LevelBadge } from '@/components/gamification/level-badge';
import { changeUserRole, removeUser } from '@/lib/admin-actions';
import { toast } from 'sonner';

interface Member {
  id: string;
  name: string | null;
  email: string;
  image: string | null;
  role: string;
  points: number;
  level: number;
  createdAt: Date;
  activeBan: {
    id: string;
    reason: string;
    expiresAt: Date;
  } | null;
}

interface MemberTableProps {
  members: Member[];
  currentUserId: string;
  currentUserRole: string;
}

export function MemberTable({ members, currentUserId, currentUserRole }: MemberTableProps) {
  const [banUserId, setBanUserId] = useState<string | null>(null);
  const [confirmRemoveId, setConfirmRemoveId] = useState<string | null>(null);
  const [isLoading, setIsLoading] = useState<string | null>(null);

  const isAdmin = currentUserRole === 'admin' || currentUserRole === 'owner';
  const isOwner = currentUserRole === 'owner';

  const canManage = (memberRole: string) => {
    const levels: Record<string, number> = { owner: 4, admin: 3, moderator: 2, member: 1 };
    return levels[currentUserRole] > levels[memberRole];
  };

  const handleRoleChange = async (userId: string, newRole: string) => {
    setIsLoading(userId);
    const result = await changeUserRole(userId, newRole);
    setIsLoading(null);

    if (result.error) {
      toast.error(result.error);
    } else {
      toast.success('Role updated');
    }
  };

  const handleRemove = async (userId: string) => {
    setIsLoading(userId);
    const result = await removeUser(userId, 'Removed by admin');
    setIsLoading(null);
    setConfirmRemoveId(null);

    if (result.error) {
      toast.error(result.error);
    } else {
      toast.success('Member removed');
    }
  };

  const banUser = banUserId ? members.find((m) => m.id === banUserId) : null;

  if (members.length === 0) {
    return (
      <p className="text-center text-muted-foreground py-8">
        No members found.
      </p>
    );
  }

  return (
    <>
      <div className="space-y-4">
        {members.map((member) => (
          <div
            key={member.id}
            className="border rounded-lg p-4"
          >
            <div className="flex items-center justify-between gap-4">
              <div className="flex items-center gap-3">
                {member.image ? (
                  <Image
                    src={member.image}
                    alt={member.name ?? 'Avatar'}
                    width={40}
                    height={40}
                    className="rounded-full"
                  />
                ) : (
                  <div className="w-10 h-10 rounded-full bg-muted flex items-center justify-center text-muted-foreground">
                    {(member.name ?? member.email)[0].toUpperCase()}
                  </div>
                )}
                <div>
                  <div className="flex items-center gap-2">
                    <span className="font-medium">{member.name ?? member.email}</span>
                    <RoleBadge role={member.role} showMember />
                    <LevelBadge level={member.level} size="sm" />
                  </div>
                  <div className="text-sm text-muted-foreground">
                    {member.email}
                  </div>
                  <div className="text-xs text-muted-foreground mt-1">
                    Joined {formatDistanceToNow(new Date(member.createdAt), { addSuffix: true })}
                    {' '}&bull;{' '}
                    {member.points} points
                  </div>
                </div>
              </div>

              {member.activeBan && (
                <div className="text-sm text-destructive bg-destructive/10 px-3 py-1.5 rounded">
                  Banned until {new Date(member.activeBan.expiresAt).toLocaleDateString()}
                </div>
              )}

              {isAdmin && member.id !== currentUserId && canManage(member.role) && (
                <div className="flex items-center gap-2">
                  {/* Role dropdown */}
                  <select
                    value={member.role}
                    onChange={(e) => handleRoleChange(member.id, e.target.value)}
                    disabled={isLoading === member.id}
                    className="text-sm border rounded px-2 py-1"
                  >
                    <option value="member">Member</option>
                    <option value="moderator">Moderator</option>
                    {isOwner && <option value="admin">Admin</option>}
                  </select>

                  {!member.activeBan && (
                    <Button
                      variant="outline"
                      size="sm"
                      onClick={() => setBanUserId(member.id)}
                    >
                      Ban
                    </Button>
                  )}

                  {confirmRemoveId === member.id ? (
                    <>
                      <Button
                        variant="destructive"
                        size="sm"
                        onClick={() => handleRemove(member.id)}
                        disabled={isLoading === member.id}
                      >
                        {isLoading === member.id ? 'Removing...' : 'Confirm'}
                      </Button>
                      <Button
                        variant="outline"
                        size="sm"
                        onClick={() => setConfirmRemoveId(null)}
                      >
                        Cancel
                      </Button>
                    </>
                  ) : (
                    <Button
                      variant="outline"
                      size="sm"
                      onClick={() => setConfirmRemoveId(member.id)}
                    >
                      Remove
                    </Button>
                  )}
                </div>
              )}
            </div>
          </div>
        ))}
      </div>

      {banUser && (
        <BanDialog
          userId={banUser.id}
          userName={banUser.name ?? banUser.email}
          onClose={() => setBanUserId(null)}
        />
      )}
    </>
  );
}

import { Metadata } from 'next';
import { redirect } from 'next/navigation';
import { getServerSession } from 'next-auth';
import { authOptions } from '@/lib/auth';
import db from '@/lib/db';
import { canManageMembers, type Role } from '@/lib/permissions';
import { MemberTable } from '@/components/admin/member-table';
import { Pagination } from '@/components/ui/pagination';

export const metadata: Metadata = {
  title: 'Member Management | Admin',
};

interface PageProps {
  searchParams: Promise<{ page?: string }>;
}

export default async function AdminMembersPage({ searchParams }: PageProps) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.id) redirect('/login');

  const currentUser = await db.user.findUnique({
    where: { id: session.user.id },
    select: { role: true },
  });

  const userRole = (currentUser?.role ?? 'member') as Role;
  if (!canManageMembers(userRole)) {
    redirect('/admin/posts?error=unauthorized');
  }

  const params = await searchParams;
  const page = Math.max(1, parseInt(params.page ?? '1', 10));
  const pageSize = 20;
  const skip = (page - 1) * pageSize;
  const now = new Date();

  const [users, total] = await Promise.all([
    db.user.findMany({
      skip,
      take: pageSize,
      orderBy: { createdAt: 'desc' },
      select: {
        id: true,
        name: true,
        email: true,
        image: true,
        role: true,
        points: true,
        level: true,
        createdAt: true,
        bans: {
          where: { expiresAt: { gt: now } },
          orderBy: { expiresAt: 'desc' },
          take: 1,
          select: {
            id: true,
            reason: true,
            expiresAt: true,
          },
        },
      },
    }),
    db.user.count(),
  ]);

  const members = users.map((user) => ({
    ...user,
    activeBan: user.bans[0] ?? null,
  }));

  const totalPages = Math.ceil(total / pageSize);

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <div>
          <h2 className="text-xl font-semibold">Member Management</h2>
          <p className="text-sm text-muted-foreground">
            {total} total members
          </p>
        </div>
      </div>

      <MemberTable
        members={members}
        currentUserId={session.user.id}
        currentUserRole={userRole}
      />

      {totalPages > 1 && (
        <Pagination
          currentPage={page}
          totalPages={totalPages}
          baseUrl="/admin/members"
        />
      )}
    </div>
  );
}
